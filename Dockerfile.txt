FROM node:20-alpine

# Instalar dependencias necesarias para compatibilidad de librerías nativas
RUN apk add --no-cache libc6-compat

# Establecer el directorio de trabajo
WORKDIR /app

# Copiar archivos de definición de dependencias
COPY package.json package-lock.json* ./

# Instalar solo dependencias de producción para optimizar el tamaño de la imagen
RUN npm install --production

# Copiar todo el código fuente del proyecto
COPY . .

# Exponer el puerto 3000 (puerto por defecto del sistema)
EXPOSE 3000

# Configurar variables de entorno para producción
ENV NODE_ENV=production
ENV PORT=3000

# Comando de inicio del nodo operativo NeuronAds
CMD ["node", "server.js"]
